# SQL Commands

If syntax is in:
[] -> its optional
() -> the value

## SET
SET (option)=(value)

option:
- NULL_TEXT
- TIMING
- MIN_LOG

## SHOW
Syntax: SHOW (ALL|option)

option:
- NULL_TEXT
- TIMING
- MIN_LOG

## RESET
Syntax: RESET (option)
option:
- NULL_TEXT
- TIMING
- MIN_LOG

## DROP TABLE
Syntax: DROP TABLE [IFEXISTS] (table)

## DROP SCHEMA
Syntax: DROP SCHEMA [IFEXISTS] [CASCADE] (schema)

## TRUNCATE
Syntax: TRUNCATE [TABLE] (table,...)

## TABLE
Syntax: TABLE (table)
